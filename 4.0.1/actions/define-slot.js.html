¶ harmony.defineSlot(opts)

Create a new adSlot in the page.

Params
opts.name String Slot name, ex) RP01
opts.id String Slot's div id, ex) ad-div-RP01
opts.sizes Array One or many 2D arrays, ex) [300, 250]
opts.adunit String Full ad unit code.
[opts.targeting] Object Slot-specific targeting.
[opts.mapping] Array Size mapping.
[opts.companion] Boolean True if companion ad.
[opts.drone] Boolean True when duplicates are anticipated.
[opts.group] String Slot group name.
[opts.interstitial] Boolean True if out-of-page ad.
[opts.enabled] Boolean False if ineligible to make ad calls.
[opts.on] Object Dictionary of callbacks.
[opts.one] Object Dictionary of single-run callbacks.
Returns
AdSlot {AdSlot}

adslot.js 
module.exports = function (opts) {
    var slot;
    try {
        slot = AdSlot(
            global.googletag.pubads(),
            Util.scrubConf(opts)
        );
        slots.add(slot);
        groups.add(opts.group, slot);
    } catch (err) {
        log('error', {
            msg: 'Slot failed to load during call to defineSlot()',
            conf: opts,
            err: err
        });
    }
    return slot;
};
