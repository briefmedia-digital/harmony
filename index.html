¶ harmony.hide.group(name)

Params
name String 
        group: function (name) {
            var i, el,
                set = groups.get(name),
                len = set.length;
            log('hide', 'Hiding ads in group ' + name);
            for (i = 0; i < len; i += 1) {
                el = document.getElementById(set[i].divId);
                if (el) {
