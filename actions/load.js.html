Generate the ad slots.

 
    var i, slot, conf,
        len = opts.slots.length;
    log('load', 'Generating ad slots.');
    for (i = 0; i < len; i += 1) {
        conf = opts.slots[i];
        try {
            slot = AdSlot(
                pubads,
                Util.scrubConf(conf)
            );
            slots.add(slot);
            groups.add(slot.group, slot);
        } catch (err) {
            log('error', {
                msg: 'Slot failed to load during call to load().',
                conf: conf,
                err: err
            });
        }
    }
